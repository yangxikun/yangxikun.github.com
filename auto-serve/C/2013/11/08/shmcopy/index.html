使用共享内存和信号量模仿cp命令

08 November 2013

学习共享内存和信号量的结合使用

学习自《linux网络编程》，以下实例程序也是仿照书上的，但是觉得书本上的实现有问题，会发生死锁，自己修改了实现方法。

实例程序shmcopy功能：把一个文件复制为另一个文件（usage:shmcopy a b）。每次调用shmcopy就形成两个进程——读进程和写进程。它们共享两个缓冲区，这两个缓冲区作为共享内存顿来实现。当读进程把数据写入第一个缓冲区期间，写进程就把第二个缓冲区内容写出去。反之亦然。由于读和写是并发进行的，所以数据吞吐量增加了。

为了使两个进程同步，防止读进程把缓冲区装满之前，写进程就把该缓冲区内容写出去。我们使用了两个信号量。因为共享存储器机构本身没有提供同步功能。

我的思路（修改了写进程）：

源代码：

  1 #ifndef __USE_FILE_OFFSET64
  2 #define __USE_FILE_OFFSET64
  3 #endif
  4 
  5 #ifndef __USE_LARGEFILE64
  6 #define __USE_LARGEFILE64
  7 #endif
  8 
  9 #ifndef _LARGEFILE64_SOURCE
 10 #define _LARGEFILE64_SOURCE
 11 #endif
 12 
 13 #include<stdio.h>
 14 #include<stdlib.h>
 15 #include<signal.h>
 16 #include<sys/types.h>
 17 #include<sys/ipc.h>
 18 #include<sys/shm.h>
 19 #include<sys/sem.h>
 20 #include<sys/stat.h>
 21 #include<fcntl.h>
 22 
 23 /* 共享内存的关键字 */
 24 #define SHMKEY1 (key_t)0x10
 25 #define SHMKEY2 (key_t)0x15
 26 /* 信号量的关键字 */
 27 #define SEMKEY (key_t)0x20
 28 
 29 /* 读写缓冲区的大小32M */
 30 #define SIZE 33554428
 31 
 32 /* 缓冲区数据结构 */
 33 struct databuf {
 34     int d_nread;
 35     char d_buf[SIZE];
 36 };
 37 
 38 #define IFLAGS (IPC_CREAT|IPC_EXCL)
 39 /* 将-1强制转换为struct databuf*类型，用于shmat执行结果的判断 */
 40 #define ERR ((struct databuf*)-1)
 41 
 42 static int shmid1,shmid2,semid;
 43 
 44 void fatal(char *mes) {
 45     perror(mes);
 46     exit(1);
 47 }
 48 /* 对共享存储器段的初始化 */
 49 void getseg(struct databuf **p1, struct databuf **p2) {
 50     if ((shmid1=shmget(SHMKEY1, sizeof(struct databuf), 0600|IFLAGS)) < 0) {
 51         fatal("shmget");
 52     }
 53     if ((shmid2=shmget(SHMKEY2, sizeof(struct databuf), 0600|IFLAGS)) < 0) {
 54         fatal("shmget");
 55     }
 56 
 57     if ((*p1=(struct databuf*)(shmat(shmid1, 0, 0))) == ERR)
 58         fatal("shmat");
 59     if ((*p2=(struct databuf*)(shmat(shmid2, 0, 0))) == ERR)
 60         fatal("shmat");
 61 }
 62 /* 对信号量的初始化 */
 63 void getsem() {
 64     if ((semid=semget(SEMKEY, 2, 0600|IFLAGS)) < 0) {
 65         fatal("semget");
 66     }
 67     if (semctl(semid, 0, SETVAL, 0) < 0) {
 68         fatal("semctl");
 69     }
 70     if (semctl(semid, 1, SETVAL, 0) < 0) {
 71         fatal("semctl");
 72     }
 73 }
 74 /* 程序结束时释放共享内存和信号量*/
 75 void removex() {
 76     if (shmctl(shmid1, IPC_RMID, 0) < 0)
 77         fatal("shmctl");
 78     if (shmctl(shmid2, IPC_RMID, 0) < 0)
 79         fatal("shmctl");
 80     if (semctl(semid, IPC_RMID, 0) < 0) {
 81         fatal("semctl");
 82     }
 83 }
 84 /* 定义对两个信号量的操作*/
 85 struct sembuf p1={0,-1,0},p2={1,-1,0},v1={0,1,0},v2={1,1,0};
 86 /* 读进程 */
 87 void reader(int semid, struct databuf *buf1, struct databuf *buf2, char *filepath) {
 88     int fd = open(filepath, O_RDONLY|O_LARGEFILE);
 89     if (fd == -1) {
 90         fatal("open");
 91     }
 92        
 93     for (;;) {
 94         buf1->d_nread = read(fd, buf1->d_buf, SIZE);
 95         semop(semid, &v1, 1);
 96         if (buf1->d_nread == -1 || buf1->d_nread == 0) {
 97             return;    
 98         }
 99         semop(semid, &p2, 1);
100 
101         buf2->d_nread = read(fd, buf2->d_buf, SIZE);
102         semop(semid, &v2, 1);
103         if (buf2->d_nread == -1 || buf2->d_nread == 0) {
104             return;
105         }
106         semop(semid, &p1, 1);
107     }
108 }
109 /* 写进程 */
110 void writer(int semid, struct databuf *buf1, struct databuf *buf2, char *filepath) {
111     int fd = open(filepath, O_WRONLY|O_CREAT|O_LARGEFILE, S_IREAD|S_IWRITE|S_IRGRP|S_IWGRP|S_IROTH|S_IWOTH);
112     if (fd == -1) {
113         fatal("open");
114     }
115 
116     for (;;) {
117         if (buf2->d_nread == -1 || buf2->d_nread == 0) {
118             return;
119         }
120         write(fd, buf2->d_buf, buf2->d_nread);
121         semop(semid, &v2, 1);
122         semop(semid, &p1, 1);
123         if (buf1->d_nread == -1 || buf1->d_nread == 0) {
124             return;
125         }
126         write(fd, buf1->d_buf, buf1->d_nread);
127         semop(semid, &v1, 1);
128         semop(semid, &p2, 1);
129 
130     }
131 }
132 /* 主进程 */
133 void main(int argc, char *argv[]) {
134     int pid;
135     struct databuf *buf1,*buf2;
136     getsem();
137 
138     getseg(&buf1, &buf2);
139     buf2->d_nread = -2;//使写进程在第一次无法将buf2中数据写入文件
140     switch(pid = fork()) {
141         case -1:fatal("fork");break;
142         case 0:writer(semid, buf1, buf2, argv[2]);break;
143         default:reader(semid, buf1, buf2, argv[1]);
144                 removex();
145     }
146     exit(0);
147 }

在测试时，发现无法读入超过2G的文件，google了下，原来32位系统C程序要打开超过2G的文件，需要一些方法：

  1. 方法一：同上面代码一样，在程序中添加宏定义，以及在打开文件时，加上O_LARGEFILE，之后使用gcc直接编译；
  2. 方法二：在gcc编译时加上一些参数：gcc -o shmcopy2 shmcopy2.c -D_FILE_OFFSET_BITS=64。

测试使用的是一个2.8G的文件，结果是性能和cp命令差不多。
