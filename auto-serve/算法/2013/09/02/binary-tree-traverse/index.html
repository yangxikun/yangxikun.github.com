二叉树的遍历

02 September 2013

存储结构

链式存储结构

1 typedef struct BiTNode{
2     TElemType data;
3     struct BiTNode *lchild,*rchild;//左右孩子指针
4 }BiTNode,*BiTree;

先序遍历(根左右)

 1 Status PreOrderTraverse( BiTree T ){
 2     InitStack( S );p=T;
 3     while( p || !StackEmpty( S ) ){
 4         if( p ){
 5             visite( p );//访问p
 6             StackPush( S, p );
 7             p = p->lchild;
 8         }else{
 9             StackPop( S, p );
10             p = p->rchild;
11         }
12     }//while
13     return OK;
14 }

中序遍历(左根右)

 1 Status InOrderTraverse( BiTree T ){
 2     InitStack( S );p=T;
 3     while( p || !StackEmpty( S ) ){
 4         if( p ){
 5             StackPush( S, p );
 6             p = p->lchild;
 7         }else{
 8             StackPop( S, p );
 9             visite( p );
10             p = p->rchild;
11         }
12     }//while
13 }

后序遍历(左右根)

 1 Status PostOrderTraverse( BiTree T ){
 2     InitStack( S );p=T;
 3     while( p || !StackEmpty( S ) ){
 4         while( p ){
 5             StackPush( S, p );
 6             p = p->lchild;
 7         }//while
 8         GetTop( S, p );//取得栈顶元素的值
 9         if( p->rchild == NULL || lastVisite == p->rchild ){
10             visite( p );
11             lastVisite = p;//记录上一次访问的结点
12             StackPop( S, p );
13             p = NULL;
14         }else{
15             p = p->rchild;
16         }
17     }//while
18     return OK;
19 }
