递归遍历目录

22 September 2013

方法一

 1 <meta charset="utf-8">
 2 <?php 
 3 function showDir( $dir )
 4 {
 5     $r = array();
 6     foreach ( scandir( $dir ) as $key => $value ) {// scandir()列出指定路径中的文件和目录
 7         if ( $value === '.' || $value === '..'  ) {
 8             continue;
 9         }
10         if ( is_dir( $dir.'/'.$value ) ) {
11             $r[$dir.'/'.$value] = showDir( $dir.'/'.$value );
12         }
13     }
14     return $r;
15 }
16 var_dump( showDir( '.' ) );
17 ?>

方法二

 1 <meta charset="utf-8">
 2 <?php
 3 function showDir( $dir )
 4 {
 5     $r              = array();
 6     $handledir = opendir( $dir ) ;
 7     while ( false !== $dirName = readdir( $handledir ) ) {//readdir()返回目录中下一个文件的文件名。文件名以在文件系统中的排序返回。 
 8         if ( $dirName==='.' || $dirName=='..' ) {
 9             continue;
10         }
11         if ( is_dir( $dir.'/'.$dirName ) ) {
12             $r[$dir.'/'.$dirName] = showDir( $dir.'/'.$dirName );
13         }
14     }
15     return $r;
16 }
17 var_dump( showDir( '.' ) );
18 ?>
