深度优先搜索

 1 Boolean visited[MAX];//访问标记数组
 2 
 3 void DFSTraverse( Graph G, int v ){
 4     for( v=0; v<G.vexnum; ++v ){
 5         visited[v] = FALSE;
 6     }
 7     for( v=0; v<G.vexnum; v++ ){
 8         if( !visited[v] )
 9             DFS( G, v );
10     }
11 }
12 
13 void DFS( Graph G, int v ){
14     visited[v] = TRUE;
15     for( w = FirstAdjVex( G, v ); w>=0; w = NextAdjVex( G, v ) )
16         if( !visited[w] )
17             DFS( G, w );
18 }

广度优先搜索

 1 void BFSTraverse( Graph G, int v ){
 2     //使用辅助队列Q和访问标记数组visited
 3     for( v=0; v<G.vexnum; v++ ){
 4         visited[v] = FALSE;
 5     }
 6     InitQueue( Q );
 7     for( v=0; v<G.vexnum; v++){
 8         if( !visited[v] ){
 9             visited[v] = TRUE;
10             EnQueue( Q, v );//入队
11             while( !QueueEmpty( Q ) ){
12                 DeQueue( Q, u );//出队
13                 for( w = FirstAdjVex( G, u ); w>=0; w = NextAdjVex( G, u ) )
14                     if( !visited[w] ){
15                         visited[w] = TRUE;
16                         EnQueue( Q, w );
17                     }//if
18             }//while
19         }//if
20     }//for
21 }
