算法代码

 1 void Knapsack(Type v, int w, int c, int n, Type **m) {
 2     int jMax = min(w[n]-1, c);
 3     for (int j = 0; j <= jMax; j++) m[n][j] = 0;
 4     for (int j = w[n]; j <= c; j++) m[n][j] = v[n];
 5     for (int i = n-1; i > 1; i--) {
 6         jMax = min(w[i]-1, c);
 7         for (int j = 0;j <= jMax; j++) m[i][j] = m[i+1][j];
 8         for (int j = w[i]; j <= c; j++) m[i][j] = max(m[i+1][j], m[i+1][j-w[i]]+v[i]);
 9     }
10     if (c >= w[1]) m[1][c] = max(m[2][c], m[2][c-w[1]]+v[1]);
11 }

13 void TrackBack(Type **m, int w, int c, int n, int x) {
14     for (int i = 1; i < n; i++)
15         if (m[i][c] == m[i+1][c])
16             x[i] = 0;
17         else {
18             x[i] = 1; 
19             c -= w[i];
20         }
21         x[n] = (m[n][c]) ? 1 : 0;
22 }
